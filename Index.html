  <script>

    // Mise à jour dynamique de l'indication de mois d'échéance

    const dateInput = document.getElementById('date_echeance');

    const moisSpan = document.getElementById('moisEcheance');

    const mois = ["janvier","février","mars","avril","mai","juin","juillet","août","septembre","octobre","novembre","décembre"];

    function updateMois(){

      if(!dateInput.value){ moisSpan.textContent = "mois du renouvellement"; return; }

      const d = new Date(dateInput.value + "T12:00:00");

      if(isNaN(d)) { moisSpan.textContent = "mois du renouvellement"; return; }

      moisSpan.textContent = mois[d.getMonth()] + " " + d.getFullYear();

    }

    dateInput.addEventListener('change', updateMois);

    updateMois();


    // Validation minimale côté client + placeholder d'envoi

    document.getElementById('soumission').addEventListener('submit', function(e){

      if(!this.checkValidity()){

        return false;

      }

        return false;

      }

    });
